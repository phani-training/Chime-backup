Amazon Chime Signaling SDK 0.1.0
Amazon Chime Signaling Client is a project that allows builders to use their custom webrtc, but use Amazon Chime signaling server. Signaling server is used to communicate information about audio and video between peers, so the signaling client will enable builders to customize their own webrtc library, but use the Amazon Chime backend.

36 static std::vector<std::string> Split(const std::string& str, const std::string& delimiter) {
37 size_t pos_start = 0, pos_end, delim_len = delimiter.length();
38 std::string token;
39 std::vector<std::string> res;
40
41 while ((pos_end = str.find(delimiter, pos_start)) != std::string::npos) {
42 token = str.substr(pos_start, pos_end - pos_start);
43 pos_start = pos_end + delim_len;
44 res.push_back(token);
45 }
46
47 res.push_back(str.substr(pos_start));
48 return res;
49 }

52 static std::vector<MediaSection> ParseSDP(const std::string& sdp) {
53 std::vector<std::string> sdp_vec = Split(sdp, "\r\n");
54 std::vector<MediaSection> media_sections;
55 const std::string mid_str = "a=mid:";
56 bool is_audio_section = false;
57 std::string mid;
58 for (const auto& line : sdp_vec) {
59 if (line.find("m=audio", 0) == 0) {
60 is_audio_section = true;
61 } else if (line.find("m=video", 0) == 0) {
62 is_audio_section = false;
63 } else if (line.find(mid_str, 0) == 0) {
64 mid = line.substr(mid_str.length(), std::string::npos);
65 } else if (line.find(rec_only, 0) == 0 || line.find(send_only, 0) == 0 || line.find(inactive, 0) == 0 ||
66 line.find(sendrecv, 0) == 0) {
67 MediaSection media_section{is_audio_section ? MediaType::kAudio : MediaType::kVideo, mid, GetDirection(line)};
68
69 media_sections.emplace_back(media_section);
70 }
71 }
72
73 return media_sections;
74 }
